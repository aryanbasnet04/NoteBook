7.WAP to check whether the entered number is prime or composite.


DECLARE SUB PRIME (N)
INPUT “ENTER ANY NUMBER”; N
CALL PRIME (N)
END

SUB PRIME (N)
C = 0
FOR I = 1 TO N
IF N MOD I = 0 THEN C = C + 1
NEXT I
IF C = 2 THEN
PRINT N; “IS PRIME NUMBER”
ELSE
PRINT N; “IS COMPOSITE NUMBER”
END IF
END SUB

8.WAP to display the reverse of the input string.


DECLARE SUB REV(N$)
CLS
INPUT “ENTER A WORD” ; W$
CALL REV(W$)
END

SUB REV(N$)
FOR I = LEN(N$) TO 1 STEP-1
C$=C$+MID$(N$,I,1)
NEXT I
PRINT “REVERSED STRING” ; C$
END SUB

9.WAP to print the string in the alternate capital.


DECLARE SUB ALTERNATE (A$)
INPUT “ENTER STRING”; A$
END

SUB ALTERNATE (A$)
FOR I= 1 TO LEN (A$)
B$= MID (A$, I, 1)
IF I MOD 2 = 0 THEN
B$= LCASE$ (B$)
ELSE
B$ = UCASE (B$)
END IF
ALT$ = ALT$ +B$
NEXT I “THE REQUIRED ALTERNATE IS” ;ALT$
END SUB
